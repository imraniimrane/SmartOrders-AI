(() => { "use strict"; // === DATA STORAGE & USER DB === // Users stored in localStorage as {username: {password, databases: {...}}} // For demo, no hashing, but in production do not store passwords plaintext. const LS_USERS_KEY = "smartorders_users"; const LS_REMEMBERED_KEY = "smartorders_remembered_user"; function getUsers() { const usersJson = localStorage.getItem(LS_USERS_KEY); return usersJson ? JSON.parse(usersJson) : {}; } function saveUsers(users) { localStorage.setItem(LS_USERS_KEY, JSON.stringify(users)); } // Current logged in user session info let currentUser = null; // === UI ELEMENTS === const authPage = document.getElementById("auth-page"); const mainApp = document.getElementById("main-app"); const loginForm = document.getElementById("login-form"); const signupForm = document.getElementById("signup-form"); const btnLoginTab = document.getElementById("btn-login"); const btnSignupTab = document.getElementById("btn-signup"); const loginUsernameInput = document.getElementById("login-username"); const loginPasswordInput = document.getElementById("login-password"); const rememberMeCheckbox = document.getElementById("remember-me"); const navDbBtn = document.getElementById("nav-db"); const navSettingsBtn = document.getElementById("nav-settings"); const btnLogout = document.getElementById("btn-logout"); const btnSwitchAccount = document.getElementById("btn-switch-account"); const databaseSelect = document.getElementById("database-select"); const dbTableBody = document.getElementById("db-table-body"); const addDataForm = document.getElementById("add-data-form"); const inputName = document.getElementById("input-name"); const inputPhone = document.getElementById("input-phone"); const inputProduct = document.getElementById("input-product"); const inputQuantity = document.getElementById("input-quantity"); const btnCreateDb = document.getElementById("btn-create-db"); const btnExport = document.getElementById("btn-export"); const btnClearDb = document.getElementById("btn-clear-db"); const exportArea = document.getElementById("export-area"); const databasePage = document.getElementById("database-page"); const settingsPage = document.getElementById("settings-page"); const themeToggle = document.getElementById("theme-toggle"); // === INITIALIZATION === function init() { // Show login form with remembered username/password if any const remembered = localStorage.getItem(LS_REMEMBERED_KEY); if (remembered) { try { const saved = JSON.parse(remembered); if (saved.username && saved.password) { loginUsernameInput.value = saved.username; loginPasswordInput.value = saved.password; rememberMeCheckbox.checked = true; } } catch { // ignore error } } setupEventListeners(); loadThemeFromStorage(); } // === EVENT LISTENERS === function setupEventListeners() { // Tab switching btnLoginTab.addEventListener("click", () => { btnLoginTab.classList.add("active"); btnSignupTab.classList.remove("active"); btnLoginTab.setAttribute("aria-pressed", "true"); btnSignupTab.setAttribute("aria-pressed", "false"); loginForm.style.display = "block"; signupForm.style.display = "none"; }); btnSignupTab.addEventListener("click", () => { btnSignupTab.classList.add("active"); btnLoginTab.classList.remove("active"); btnSignupTab.setAttribute("aria-pressed", "true"); btnLoginTab.setAttribute("aria-pressed", "false"); signupForm.style.display = "block"; loginForm.style.display = "none"; }); // Login submit loginForm.addEventListener("submit", e => { e.preventDefault(); const username = loginUsernameInput.value.trim(); const password = loginPasswordInput.value; const remember = rememberMeCheckbox.checked; const users = getUsers(); if (users[username] && users[username].password === password) { currentUser = username; showMainApp(); if (remember) { localStorage.setItem(LS_REMEMBERED_KEY, JSON.stringify({ username, password })); } else { localStorage.removeItem(LS_REMEMBERED_KEY); } loadUserData(); } else { alert("Invalid username or password"); } }); // Signup submit signupForm.addEventListener("submit", e => { e.preventDefault(); const username = document.getElementById("signup-username").value.trim(); const password = document.getElementById("signup-password").value; if (!username || !password) { alert("Please enter username and password"); return; } const users = getUsers(); if (users[username]) { alert("Username already exists"); return; } users[username] = { password: password, databases: {} }; saveUsers(users); alert("Account created! You can now log in."); btnLoginTab.click(); }); // Logout & Switch Account btnLogout.addEventListener("click", logoutUser); btnSwitchAccount.addEventListener("click", logoutUser); // Navigation buttons navDbBtn.addEventListener("click", () => { navDbBtn.classList.add("active"); navDbBtn.setAttribute("aria-pressed", "true"); navSettingsBtn.classList.remove("active"); navSettingsBtn.setAttribute("aria-pressed", "false"); databasePage.style.display = "block"; settingsPage.style.display = "none"; }); navSettingsBtn.addEventListener("click", () => { navSettingsBtn.classList.add("active"); navSettingsBtn.setAttribute("aria-pressed", "true"); navDbBtn.classList.remove("active"); navDbBtn.setAttribute("aria-pressed", "false"); databasePage.style.display = "none"; settingsPage.style.display = "block"; }); // Create new database btnCreateDb.addEventListener("click", () => { const dbName = prompt("Enter new database name:"); if (!dbName) return; addNewDatabase(dbName.trim()); }); // Change selected database databaseSelect.addEventListener("change", renderDatabaseTable); // Add new data entry addDataForm.addEventListener("submit", e => { e.preventDefault(); addDataEntry(); }); // Export data btnExport.addEventListener("click", exportCurrentDatabase); // Clear database btnClearDb.addEventListener("click", () => { if (confirm("Are you sure you want to clear this database? This cannot be undone.")) { clearCurrentDatabase(); } }); // Theme toggle themeToggle.addEventListener("change", () => { const darkOn = themeToggle.checked; setTheme(darkOn); localStorage.setItem("smartorders_theme_dark", darkOn ? "1" : "0"); }); } // === LOGIN & USER DATA === function showMainApp() { authPage.style.display = "none"; mainApp.style.display = "block"; } function showAuthPage() { authPage.style.display = "block"; mainApp.style.display = "none"; } function logoutUser() { currentUser = null; showAuthPage(); clearDatabaseUI(); } function loadUserData() { const users = getUsers(); if (!currentUser || !users[currentUser]) { alert("User data missing, logging out."); renderDatabaseTable(); } function addNewDatabase(name) { if (!name) return; const users = getUsers(); if (!users[currentUser]) return; if (users[currentUser].databases[name]) { alert("Database with this name already exists."); return; } users[currentUser].databases[name] = []; saveUsers(users); loadUserData(); } function renderDatabaseTable() { const users = getUsers(); if (!users[currentUser]) return; const dbName = databaseSelect.value; const dbData = users[currentUser].databases[dbName] || []; dbTableBody.innerHTML = ""; if (dbData.length === 0) { const tr = document.createElement("tr"); const td = document.createElement("td"); td.colSpan = 4; td.style.textAlign = "center"; td.textContent = "No entries found."; tr.appendChild(td); dbTableBody.appendChild(tr); return; } dbData.forEach((entry, idx) => { const tr = document.createElement("tr"); tr.setAttribute("data-idx", idx); tr.innerHTML = ` ${entry.name} ${entry.phone} ${entry.product} ${entry.quantity} `; dbTableBody.appendChild(tr); }); // Attach delete handlers dbTableBody.querySelectorAll(".btn-delete").forEach(btn => { btn.addEventListener("click", e => { const idx = parseInt(e.target.dataset.idx, 10); if (!users[currentUser].databases[dbName]) return; users[currentUser].databases[dbName].push({ name, phone, product, quantity, }); saveUsers(users); renderDatabaseTable(); // Reset form addDataForm.reset(); if (!users[currentUser]) return; const dbName = databaseSelect.value; if (!users[currentUser].databases[dbName]) return; users[currentUser].databases[dbName] = []; saveUsers(users); renderDatabaseTable(); } function clearDatabaseUI() { databaseSelect.innerHTML = ""; dbTableBody.innerHTML = ""; } function exportCurrentDatabase() { const users = getUsers(); if (!users[currentUser]) return; const dbName = databaseSelect.value; exportArea.style.display = "block"; exportArea.select(); } // === THEME HANDLING === function setTheme(darkModeOn) { if (darkModeOn) { document.body.classList.add("dark-mode"); themeToggle.setAttribute("aria-checked", "true"); } else { document.body.classList.remove("dark-mode"); themeToggle.setAttribute("aria-checked", "false"); } } function loadThemeFromStorage() { const savedTheme = localStorage.getItem("smartorders_theme_dark"); const darkOn = savedTheme === "1"; themeToggle.checked = darkOn; setTheme(darkOn); } // Start the app init(); })();
